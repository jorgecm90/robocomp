107  // Refills laserData with new values
108  RoboCompLaser::TLaserData LASER_createLaserData(const IMVLaser &laser)
109  {
110  QMutexLocker vm(worker->viewerMutex);
111  QMutexLocker locker(worker->mutex);
112  QMutexLocker ldc(laserDataCartArray_mutex);
113 // printf("osg threads running... %d\n", viewer->areThreadsRunning());
114  static RoboCompLaser::TLaserData laserData;
115  int measures = laser.laserNode->measures;
116  QString id = laser.laserNode->id;
117  float iniAngle = -laser.laserNode->angle/2;
118  float finAngle = laser.laserNode->angle/2;
119  float_t maxRange = laser.laserNode->max;
120  laserData.resize(measures);
121 
122  double angle = finAngle; //variable to iterate angle increments
123  //El punto inicial es el origen del láser
124  const osg::Vec3 P = QVecToOSGVec(innerModel->laserTo("root", id, 0, 0));
125  const float incAngle = (fabs(iniAngle)+fabs(finAngle)) / (float)measures;
126  osg::Vec3 Q,R;
127 
128 
129  for (int i=0 ; i<measures; i++)
130  {
131  laserData[i].angle = angle;
132  laserData[i].dist = maxRange;
133 
134 
135  laserDataCartArray[id]->operator[](i) = QVecToOSGVec(QVec::vec3(maxRange*sin(angle), 0, maxRange*cos(angle)));
136 
137  //Calculamos el punto destino
138  Q = QVecToOSGVec(innerModel->laserTo("root", id, maxRange, angle));
139  //Creamos el segmento de interseccion
140  osg::ref_ptr<osgUtil::LineSegmentIntersector> intersector = new osgUtil::LineSegmentIntersector(osgUtil::Intersector::MODEL, P, Q);
141  osgUtil::IntersectionVisitor visitor(intersector.get());
142 
144  viewer->getRootGroup()->accept(visitor);
145 
146  if (intersector->containsIntersections() and id!="laserSecurity")
147  {
148  osgUtil::LineSegmentIntersector::Intersection result = *(intersector->getIntersections().begin());
149  R = result.getWorldIntersectPoint(); // in world space
150 
151  R.x() = R.x() - P.x();
152  R.y() = R.y() - P.y();
153  R.z() = R.z() - P.z();
154  const float dist = sqrt(R.x() *R.x() + R.y() *R.y() + R.z() *R.z());
155 
156  if (dist <= maxRange)
157  {
158  laserData[i].dist = dist;//*1000.;
159  laserDataCartArray[id]->operator[](i) = QVecToOSGVec(innerModel->laserTo(id, id, dist, laserData[i].angle));
160  }
161  }
162  else
163  {
164  laserDataCartArray[id]->operator[](i) = QVecToOSGVec(innerModel->laserTo(id, id, maxRange, laserData[i].angle));
165  }
166  angle -= incAngle;
167  }
168  // the point of the laser robot
169 // laserDataCartArray[id]->operator[](measures) = QVecToOSGVec(innerModel->laserTo(id, id, 0.0001, 0.001));
170 // viewer->startThreading();
171  return laserData;
172  }

444  // Update all the joint positions
445  void updateJoints(const float delta)
446  {
447 // printf("%s: %d\n", __FILE__, __LINE__);
448  QHash<QString, JointMovement>::const_iterator iter;
449  for (iter = jointMovements.constBegin() ; iter != jointMovements.constEnd() ; ++iter)
450  {
451  InnerModelNode *node = innerModel->getNode(iter.key());
452  InnerModelJoint *ajoint;
453  InnerModelPrismaticJoint *pjoint;
454  if ((ajoint = dynamic_cast<InnerModelJoint*>(node)) != NULL)
455  {
456  const float angle = ajoint->getAngle();
457  const float amount = fminf(fabsf(iter->endPos - angle), iter->endSpeed *delta);
458  switch (iter->mode)
459  {
460  case JointMovement::FixedPosition:
461  ajoint->setAngle(iter->endPos);
462  break;
463  case JointMovement::TargetPosition:
464  if (iter->endPos > angle)
465  ajoint->setAngle(angle + amount);
466  else if (iter->endPos < angle)
467  ajoint->setAngle(angle - amount);
468  break;
469  case JointMovement::TargetSpeed:
470  ajoint->setAngle(angle + iter->endSpeed *delta);
471  break;
472  default:
473  break;
474  }
475  }
476  else if ((pjoint = dynamic_cast<InnerModelPrismaticJoint*>(node)) != NULL)
477  {
478  pjoint->setPosition(iter->endPos);
479  }
480  }
481 // printf("%s: %d\n", __FILE__, __LINE__);
482  }

840 void SpecificWorker::compute()
841 {
842  // Compute the elapsed time interval since the last update
843  static QTime lastTime = QTime::currentTime();
844  QTime currentTime = QTime::currentTime();
845  const int elapsed = lastTime.msecsTo (currentTime);
846 // printf("elapsed %d\n", elapsed);
847  lastTime = currentTime;
848 
849  QMutexLocker locker(mutex);
850 
851 
852 // // Remove previous laser shapes
853  for (QHash<QString, IMVLaser>::iterator laser = d->imv->lasers.begin(); laser != d->imv->lasers.end(); laser++)
854  {
855  QMutexLocker locker(viewerMutex);
856  if (laser->osgNode->getNumChildren() > 0)
857  {
858  laser->osgNode->removeChild(0, laser->osgNode->getNumChildren());
859  }
860  }
861 
862  // Camera render
863  QHash<QString, IMVCamera>::const_iterator i = d->imv->cameras.constBegin();
864  {
865  QMutexLocker vm(viewerMutex);
866 
867  while (i != d->imv->cameras.constEnd())
868  {
869  RTMat rt= d->innerModel->getTransformationMatrix("root",i.key());
870  // Put camera in its position
871  d->imv->cameras[i.key()].viewerCamera->getCameraManipulator()->setByMatrix(QMatToOSGMat4(rt));
872 
873  for (int n=0; n<d->imv->cameras.size() ; ++n)
874  {
875  d->imv->cameras[i.key()].viewerCamera->frame();
876  }
877  i++;
878  }
879  }
880  // Laser
881  {
882  QMutexLocker vm(viewerMutex);
883  QMutexLocker lcds(d->laserDataCartArray_mutex);
884  for (QHash<QString, IMVLaser>::iterator laser = d->imv->lasers.begin(); laser != d->imv->lasers.end(); laser++)
885  {
886  QString id=laser->laserNode->id;
887 
888  if (d->laserDataCartArray.contains(id) == false)
889  {
890  //laserDataCartArray.insert(id);
891  osg::Vec3Array *v= new osg::Vec3Array();
892  v->resize(laser->laserNode->measures+1);
893  d->laserDataCartArray.insert(id,v);
894  }
895 
896  // create and insert laser data
897  d->worker = this;
898  d->laserDataArray.insert(laser->laserNode->id, d->LASER_createLaserData(laser.value()));
899 
900  // create and insert laser shape
901  osg::ref_ptr<osg::Node> p=NULL;
902  if (id=="laserSecurity")
903  {
904  p = d->viewer->addPolygon(*(d->laserDataCartArray[id]), osg::Vec4(0.,0.,1.,0.4));
905  }
906  else
907  {
908  p = d->viewer->addPolygon(*(d->laserDataCartArray[id]));
909  }
910  if (p!=NULL)
911  {
912  laser->osgNode->addChild(p);
913  }
914  //printf("%d (%d)\n", i, __LINE__);
915  }
916  }
917 
918 #ifdef INNERMODELMANAGERDEBUG
919  printf("Elapsed time: %d\n", elapsed);
920 #endif
921  // Update joints and compute physic interactions
922  d->updateJoints(float(elapsed)/1000.0f);
923 
924  // Update touch sensors
925  d->updateTouchSensors();
926 
927 
928  // Shutdown empty servers
929  for (int i=0; i<d->jointServersToShutDown.size(); i++)
930  {
931  d->jointServersToShutDown[i]->shutdown();
932  }
933  d->jointServersToShutDown.clear();
934 
935  // Resize world widget if necessary, and render the world
936  {
937  QMutexLocker vm(viewerMutex);
938  if (d->viewer->size() != frameOSG->size())
939  {
940  d->viewer->setFixedSize(frameOSG->width(), frameOSG->height());
941  }
942  d->innerModel->update();
943  d->imv->update();
944  //osg render

946  }
