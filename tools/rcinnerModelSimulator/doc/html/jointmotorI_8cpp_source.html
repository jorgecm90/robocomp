52 void JointMotorI::add(QString id)
53 {
54  jointIDs << id;
55 
56  MotorParams param;
57  param.invertedSign = false;
58  param.busId = jointIDs.size();
59 
60  InnerModelNode *node = innerModel->getNode(id);
61  if (dynamic_cast<InnerModelJoint*>(node) != NULL)
62  {
63  param.minPos = dynamic_cast<InnerModelJoint*>(node)->min;
64  param.maxPos = dynamic_cast<InnerModelJoint*>(node)->max;
65  }
66  else if (dynamic_cast<InnerModelPrismaticJoint*>(node) != NULL)
67  {
68  param.minPos = dynamic_cast<InnerModelPrismaticJoint *>(node)->min;
69  param.maxPos = dynamic_cast<InnerModelPrismaticJoint *>(node)->max;
70  }
71 
72  param.maxVelocity = 10000.;
73  param.zeroPos = 0.;
74  param.stepsRange = 0;
75  param.maxDegrees = 0;
76  param.name = id.toStdString();
77  params.push_back(param);
78 
79  MotorState state;
80  state.p = 0;
81  state.v = 0;
82  state.temperature = 0;
83  state.isMoving = false;
84  state.pos = 0.;
85  state.vel = 0.;
86  state.power = 0.;
87 // state.timeStamp;
88  states[id.toStdString()] = state;
89 
90  busparams.numMotors++;
91 }
