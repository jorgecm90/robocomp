123 #define MILIMETERS_PER_UNIT 1.
124 void DifferentialRobotI::updateInnerModelPose(bool force)
125 {
126  if ( (fabs(advVel)< 0.0001 and fabs(rotVel)< 0.0001) and not force)
127  {
128  return;
129  }
130 
131  timeval now;
132  gettimeofday(&now, NULL);
133  const double msecs = (now.tv_sec - lastCommand_timeval.tv_sec)*1000. +(now.tv_usec - lastCommand_timeval.tv_usec)/1000.;
134  lastCommand_timeval = now;
135  double Angle = 0.;
136  QVec newPos, noisyNewPos;
137 
138  const double noise = (nodeOmni==NULL)?node->noise:nodeOmni->noise;
139  const QString id = (nodeOmni==NULL)?node->id:nodeOmni->id;
140 
141  // With random noise:
142  QVec rndmRot = QVec::gaussianSamples(1, 0, noise);
143  QVec rndmAdv = QVec::gaussianSamples(1, 0, noise);
144  QVec rndmYaw = QVec::gaussianSamples(1, 0, noise*0.01);
145 
146 
147  double Ax1, Az1;
148  double Ax2, Az2;
149  if (fabs(rotVel) < 0.0001)
150  {
151  // With noise:
152  if (fabs(noise)>0.0000001)
153  {
154  Ax1 = (advVel*rndmRot[0])*msecs / 1000.;
155  Az1 = (advVel+(rndmAdv[0]))*msecs / (1000.*MILIMETERS_PER_UNIT);
156  noisyNewAngle += rndmYaw[0];
157  }
158  else
159  {
160  Ax1 = 0;
161  Az1 = (advVel*msecs) / (1000.*MILIMETERS_PER_UNIT);
162  }
163  // Without noise:
164  Ax2 = 0;
165  Az2 = (advVel*msecs) / (1000.*MILIMETERS_PER_UNIT);
166  }
167  else
168  {
169  double T;
170  // With noise:
171  if (fabs(noise)>0.0000001)
172  {
173  T = advVel*msecs / 1000.;
174  Angle = (msecs * (rotVel+(1.+rndmYaw[0]))) / 1000.;
175  Ax1 = ((1.-cos(Angle))/Angle)*T*(1.+(rndmRot[0]));
176  Az1 = (sin(Angle)/Angle)*T*(1.+(rndmAdv[0]));
177  }
178  else
179  {
180  T = advVel*msecs / 1000.;
181  Angle = msecs * rotVel / 1000.;
182  Ax1 = ((1.-cos(Angle))/Angle)*T;
183  Az1 = (sin(Angle)/Angle)*T;
184  }
185  // Without noise
186  T = advVel*msecs / 1000.;
187  Angle = msecs * rotVel / 1000.;
188  Ax2 = ((1.-cos(Angle))/Angle)*T;
189  Az2 = (sin(Angle)/Angle)*T;
190  newAngle += Angle;
191  }
192 
193  QVec backNoisyNewPos = innerModel->transform(parent->id, QVec::vec3(0,0,0), id);
194  float backNoisyAngle = noisyNewAngle;
195  noisyNewAngle += Angle+((rndmYaw[0]));
196  noisyNewPos = innerModel->transform(parent->id, QVec::vec3(Ax1, 0, Az1), id);
197  innerModel->updateTransformValues(id, noisyNewPos(0), noisyNewPos(1), noisyNewPos(2), 0, noisyNewAngle, 0);
198  if (canMoveBaseTo(id, noisyNewPos, noisyNewAngle+Angle+(rndmYaw[0]*noise) ))
199  {
200  // Noisy pose(real)
201  pose.x = noisyPose.x = noisyNewPos(0)*MILIMETERS_PER_UNIT;
202  pose.z = noisyPose.z = noisyNewPos(2)*MILIMETERS_PER_UNIT;
203  pose.alpha = noisyPose.alpha = noisyNewAngle;
204  }
205  else
206  {
207  noisyNewAngle = backNoisyAngle;
208  noisyNewPos = backNoisyNewPos;
209  innerModel->updateTransformValues(id, noisyNewPos(0), noisyNewPos(1), noisyNewPos(2), 0, noisyNewAngle, 0);
210  }
211  newPos = innerModel->transform(parent->id, QVec::vec3(Ax2, 0, Az2), id+"_odometry\"");
212  innerModel->updateTransformValues(id+"_odometry\"", newPos(0), newPos(1), newPos(2), 0, newAngle, 0);
213 
214  // Pose without noise (as if I moved perfectly)
215  pose.correctedX = newPos(0)*MILIMETERS_PER_UNIT;
216  pose.correctedZ = newPos(2)*MILIMETERS_PER_UNIT;
217  pose.correctedAlpha = newAngle;
218 }

220 bool DifferentialRobotI::canMoveBaseTo(const QString nodeId, const QVec position, const double alpha)
221 {
222  if (not node->collide) return true;
223 
224  std::vector<QString> robotNodes;
225  std::vector<QString> restNodes;
226 
227  recursiveIncludeMeshes(innerModel->getRoot(), nodeId, false, robotNodes, restNodes);
228 
229  for (uint32_t in=0; in<robotNodes.size(); in++)
230  {
231  for (uint32_t out=0; out<restNodes.size(); out++)
232  {
233  if (innerModel->collide(robotNodes[in], restNodes[out]))
234  {
235  return false;
236  }
237  }
238  }
239 
240  return true;
241 }

243 void DifferentialRobotI::recursiveIncludeMeshes(InnerModelNode *node, QString robotId, bool inside, std::vector<QString> &in, std::vector<QString> &out)
244 {
245  if (node->id == robotId)
246  {
247  inside = true;
248  }
249 
250  InnerModelMesh *mesh;
251  InnerModelPlane *plane;
252  InnerModelTransform *transformation;
253 
254  if ((transformation = dynamic_cast<InnerModelTransform *>(node)))
255  {
256  for (int i=0; i<node->children.size(); i++)
257  {
258  recursiveIncludeMeshes(node->children[i], robotId, inside, in, out);
259  }
260  }
261  else if ((mesh = dynamic_cast<InnerModelMesh *>(node)) or (plane = dynamic_cast<InnerModelPlane *>(node)))
262  {
263  if (node->collidable)
264  {
265  //printf("collidable: %s\n", node->id.toStdString().c_str());
266  if (inside)
267  {
268  in.push_back(node->id);
269  }
270  else
271  {
272  out.push_back(node->id);
273  }
274  }
275  }
