95 #define MILIMETERS_PER_UNIT 1.
96 void OmniRobotI::updateInnerModelPose(bool force)
97 {
98  if ( (fabs(advVelx)<0.0001 and fabs(advVelz)<0.0001 and fabs(rotVel)<0.0001) and not force)
99  {
100  return;
101  }
102 
103  timeval now;
104  gettimeofday(&now, NULL);
105  const double msecs = (now.tv_sec - lastCommand_timeval.tv_sec)*1000. +(now.tv_usec - lastCommand_timeval.tv_usec)/1000.;
106  lastCommand_timeval = now;
107 
108  QVec vel = QVec::vec3(advVelx, 0, advVelz);
109  QVec newPos, noisyNewPos;
110  const double noise = node->noise;
111 
112  // Random noise:
113  QVec rndmPos = QVec::gaussianSamples(2, 0, msecs*noise*(0.001 *vel.norm2() + 0.01 *rotVel));
114  QVec rndmYaw = QVec::gaussianSamples(1, 0, msecs*noise*(0.0001*vel.norm2() + 0.001*rotVel));
115 
116  // Without noise
117  QVec T = vel.operator*(msecs / 1000.);
118  float Angle = rotVel*msecs / 1000.;
119  newAngle += Angle;
120 
121  QVec backNoisyNewPos = innerModel->transform(parent->id, QVec::vec3(0,0,0), node->id);
122  float backNoisyAngle = noisyNewAngle;
123  noisyNewAngle += Angle + rndmYaw[0];
124  noisyNewPos = innerModel->transform(parent->id, QVec::vec3(T(0), 0, T(2)), node->id);
125  innerModel->updateTransformValues(node->id, noisyNewPos(0), noisyNewPos(1), noisyNewPos(2), 0, noisyNewAngle, 0);
126  if (canMoveBaseTo(node->id, noisyNewPos, noisyNewAngle+Angle+(rndmYaw[0]*noise) ))
127  {
128  // Noisy pose(real)
129  pose.x = noisyPose.x = noisyNewPos(0)*MILIMETERS_PER_UNIT;
130  pose.z = noisyPose.z = noisyNewPos(2)*MILIMETERS_PER_UNIT;
131  pose.alpha = noisyPose.alpha = noisyNewAngle;
132  }
133  else
134  {
135  noisyNewAngle = backNoisyAngle;
136  noisyNewPos = backNoisyNewPos;
137  innerModel->updateTransformValues(node->id, noisyNewPos(0), noisyNewPos(1), noisyNewPos(2), 0, noisyNewAngle, 0);
138  }
139  newPos = innerModel->transform(parent->id, QVec::vec3(T(0)+rndmPos[0], 0, T(2)+rndmPos[1]), node->id+"_odometry\"");
140  innerModel->updateTransformValues(node->id+"_odometry\"", newPos(0), newPos(1), newPos(2), 0, newAngle, 0);
141 
142  // Pose without noise (as if I moved perfectly)
143  pose.correctedX = newPos(0)*MILIMETERS_PER_UNIT;
144  pose.correctedZ = newPos(2)*MILIMETERS_PER_UNIT;
145  pose.correctedAlpha = newAngle;
146 }

175 void OmniRobotI::recursiveIncludeMeshes(InnerModelNode *node, QString robotId, bool inside, std::vector<QString> &in, std::vector<QString> &out)
176 {
177  if (node->id == robotId)
178  {
179  inside = true;
180  }
181 
182  InnerModelMesh *mesh;
183  InnerModelPlane *plane;
184  InnerModelTransform *transformation;
185 
186  if ((transformation = dynamic_cast<InnerModelTransform *>(node)))
187  {
188  for (int i=0; i<node->children.size(); i++)
189  {
190  recursiveIncludeMeshes(node->children[i], robotId, inside, in, out);
191  }
192  }
193  else if ((mesh = dynamic_cast<InnerModelMesh *>(node)) or (plane = dynamic_cast<InnerModelPlane *>(node)))
194  {
195  if (inside)
196  {
